<script>
	(function(){
		var ajaxSending = false,
			url = "http://imas.gree-apps.net/app/index.php/event/350/raid/getEventIdolForAjax",
			$elemSelectedIdol = $('#event-raid-g-selected-idol'),
			eventIdolInfo = _.template($('#tmpl-event-idol-info').text()),
			loaderImg = "http://cdn.bn765.com/24b/796e725dea7fa398a87c6aa1246cba93c30ab582d68b9c5869abdd4c08ccc54c?5935b0606dd7c850ce1a2c8808fa5571";

		var eventIdol = {
			viewTemplate: function(elem, data) {
				$('.dialog-placeholder').html(elem(data));
			},

			request: function(data) {
				//Ajaxリクエストの処理

				if(ajaxSending) {
					return false;
				}

				ajaxSending = $.ajax({
					'url': url,
					'data': data,
					'dataType': 'json',
					success: function(json) {
						//成功した時の処理 データのバインドを書くことになります
						eventIdol.viewTemplate(eventIdolInfo, json);
					},
					error: function(xhr) {
						//失敗した時の処理 エラーが出た旨をユーザーに伝える処理を書きます
					},
					complete: function() {
						ajaxSending = false;
					}
				});
			}
		};

		//ここからクリックした時の処理を書く
		$('.img-horizontal li').on('click', function(){
			var idolId = $(this).data('idol-id');
			eventIdol.request({ 'idol_id':idolId });
			$('.dialog-placeholder').html('<div id="theater-loading" class="theater-loading"><img src="' + loaderImg + '"></div>');
			$('#dialog').addClass('is-show');
			$('body').addClass('dialog-open').append('<div class="overlay overlay-black"></div>');
		});

		 $('.dialog-close').on('click', function(){
			$('#dialog').removeClass('is-show');
			$('body').removeClass('dialog-open');
			$('.overlay').remove();
		});

		$(document).on('submit', '#set-idol', function(){
			$('.js-invalid-double-click').removeClass('base-btn').addClass('base-btn-off').prop('disabled', 'disabled');
		});

	})();

</script>
